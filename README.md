- src/
  - entities/
    - User.js
    - Product.js
  - factories/
    - UserFactory.js
    - ProductFactory.js
  - repositories/
    - UserRepository.js
    - ProductRepository.js
  - routes/
    - UserRoute.js
    - ProductRoute.js
  - services/
    - UserService.js
    - ProductService.js
  - util/
    - Validation.js
    - ErrorHandling.js
  - handler/
    - RouteHandler.js
- tests/
  - integration/
    - UserIntegrationTest.js
    - ProductIntegrationTest.js
